## for unix based system
sudo apt update
sudo apt install -y tesseract-ocr poppler-utils

## for windows
# Download the Tesseract installer from https://github.com/UB-Mannheim/tesseract/wiki
# Download the Poppler for Windows from https://github.com/oschwartz10612/poppler-windows/releases/
# Add the Tesseract and Poppler bin directories to your system PATH environment variable
# Restart your command prompt or IDE to apply the changes
# Verify the installation by running the following commands in your command prompt or terminal:
# tesseract --version
# pdftoppm -v
